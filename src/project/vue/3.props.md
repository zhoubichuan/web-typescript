## 复杂类型

```ts
{
        type: Array as () => (number | string)[],
        default: () => []
    }
```

```ts
{
        type: Object as () => Node,
        require: true,
        validator(items: Node) {
            return JSON.stringify(items) != '{}'
        }
    },
```
